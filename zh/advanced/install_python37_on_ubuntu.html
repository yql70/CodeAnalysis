# еңЁ Ubuntu е®үиЈ… Python3.7

жіЁпјҡеҪ“еүҚUbuntuзүҲжң¬дёә18.04

# дёӢиҪҪPythonжәҗз ҒеҢ…

wget https://www.python.org/ftp/python/3.7.12/Python-3.7.12.tgz

# е®үиЈ…еүҚеҮҶеӨҮ

е®үиЈ…дҫқиө–з»„д»¶

apt-get update
apt-get install build-essential zlib1g-dev libncurses5-dev libgdbm-dev libnss3-dev libssl-dev libsqlite3-dev libreadline-dev libffi-dev wget libbz2-dev tk-dev gcc make

# и§ЈеҺӢе®үиЈ…

# и§ЈеҺӢеҲ°/usr/local/srcзӣ®еҪ•
$ tar zvxf Python-3.7.12.tgz -C /usr/local/src
$ cd /usr/local/src/Python-3.7.12
# зј–иҜ‘еүҚй…ҚзҪ®
$ ./configure prefix=/usr/local/python3 --enable-shared
# зј–иҜ‘жһ„е»ә
$ make -j8
# е®үиЈ…Python
$ make install
# жё…зҗҶзј–иҜ‘дә§еҮәзҡ„дёӯй—ҙж–Үд»¶
$ make clean
# й“ҫжҺҘжһ„е»әдә§еҮәзҡ„PythonеҸҜжү§иЎҢж–Үд»¶еҲ°/usr/local/binзӣ®еҪ•
$ ln -s /usr/local/python3/bin/python3 /usr/local/bin/python
# й“ҫжҺҘжһ„е»әдә§еҮәзҡ„pip3еҸҜжү§иЎҢж–Үд»¶еҲ°/usr/local/binзӣ®еҪ•
$ ln -s /usr/local/python3/bin/pip3 /usr/local/bin/pip
# й“ҫжҺҘжһ„е»әдә§еҮәзҡ„PythonеҠЁжҖҒеә“
$ ln -s /usr/local/python3/lib/libpython3.7m.so.1.0 /usr/lib/libpython3.7m.so.1.0
# й…ҚзҪ®еҠЁжҖҒеә“
$ ldconfig

# жЈҖжҹҘ

жЈҖжҹҘPythonзүҲжң¬жҳҜеҗҰе®үиЈ…жҲҗеҠҹ

$ python --version
Python 3.7.12  # жӯЈеёёиҫ“еҮәпјҢиЎЁзӨәе®үиЈ…жҲҗеҠҹ

жіЁпјҡ

  1. й“ҫжҺҘеҲ°/usr/local/bin/зӣ®еҪ•дёҚдјҡеҪұе“Қзі»з»ҹиҪҜд»¶
  2. дёҖиҲ¬жғ…еҶөдёӢпјҢPATHй…ҚзҪ®жҳҜе…Ҳ/usr/local/binеҶҚ/usr/bin
  3. жЈҖжҹҘpython -vиҫ“еҮәз»“жһңжҳҜеҗҰдёәPython 3.7.12зүҲжң¬пјҢеҰӮжһңдёҚжҳҜиҜҘзүҲжң¬пјҢеҸҜиғҪеҪұе“ҚеҗҺз»ӯдҫқиө–е®үиЈ…е’ҢжңҚеҠЎиҝҗиЎҢ

# pypiдёӢиҪҪжәҗй…ҚзҪ®

pipй»ҳи®ӨжҳҜеҲ°pypiе®ҳж–№жәҗдёӢиҪҪз¬¬дёүж–№дҫқиө–еҢ…пјҢдёӢиҪҪйҖҹеәҰеҸҜиғҪдјҡжҜ”иҫғж…ўпјҢеҸҜд»ҘиҖғиҷ‘и°ғж•ҙдёәи…ҫи®Ҝдә‘зҡ„pypiдёӢиҪҪжәҗпјҢи°ғж•ҙж–№ејҸпјҡ

mkdir ~/.pip/
echo "[global]\nindex-url = https://mirrors.cloud.tencent.com/pypi/simple" >> ~/.pip/pip.conf
